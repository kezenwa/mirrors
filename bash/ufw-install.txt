## configure ufw rules ##
ufw default deny incoming
ufw default allow outgoing
ufw allow 6969
ufw allow 80
ufw allow 443
ufw --force disable
ufw --force enable
echo "y" | ufw enable
/etc/init.d/ufw force-reload
/etc/init.d/ufw restart
